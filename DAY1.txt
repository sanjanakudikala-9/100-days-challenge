// PRINT ALTERNATE ELEMENTS 
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter size");
        int n = sc.nextInt();
        int arr[] = new int[n];
        for(int i=0;i<n;i++){
            arr[i] = sc.nextInt();
        }
        printAlternative(arr);
    }
    private static void printAlternative(int arr[]){
        System.out.println("Even alternatives :");
        for(int i=0; i<arr.length;i+=2){
            System.out.println(arr[i]+" ");
        }
        System.out.println("odd alternatives :");
        for(int i=1; i<arr.length-1;i+=2){
            System.out.println(arr[i]+" ");
        }
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter size");
        int n = sc.nextInt();
        int arr[] = new int[n];
        for(int i=0;i<n;i++){
            arr[i] = sc.nextInt();
        }
        ArrayList<Integer> res = printLeaders(arr);
        System.out.print(res);
    }
    private static ArrayList<Integer> printLeaders(int arr[]){
        ArrayList<Integer> list = new ArrayList<>();
        int n = arr.length;
        for(int i=0; i<arr.length;i++){
            int j;
            for(j=i+1;j<arr.length;j++){
                if(arr[i]<arr[j]){
                    break;
                    // if the condition fails then there itself the traversal stops 
                    // if there exist any element on its right which is greater than or equal 
                    // instead of checking greater one we did in reverse direction
                    // we checked whether curr is less than the ele which on the right side if so then break 
                    // else continue the traversal
                }
            }
            // here if the length of j is equal to length of the arr that concludes that the traversal 
            // of the array is completed if it breaks in the middle then the element is added to the list 
                if(j==n){
                    list.add(arr[i]);
            }
            
        }
        return list;
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter size");
        int n = sc.nextInt();
        int arr[] = new int[n];
        for(int i=0;i<n;i++){
            arr[i] = sc.nextInt();
        }
        ArrayList<Integer> res = printLeaders(arr);
        System.out.print(res);
    }
    private static ArrayList<Integer> printLeaders(int arr[]){
        ArrayList<Integer> list = new ArrayList<>();
        int n = arr.length;
        int maxRightEle = arr[n-1];
        // we started traversal from rightmost side so there will be no ele gretaer than that as it is the
        // last element so we add it to the list initially 
        list.add(maxRightEle);
        // and start traversal from the 2nd ele as it is 0-based index we use n-2 
        // if the ele is greater than the rightmost ele then we change maxRightEle and add it to the list 
        for(int i=n-2;i>=0;i--){
            if(arr[i]>=maxRightEle){
                maxRightEle = arr[i];
                list.add(maxRightEle);
            }
        }
        // while adding the insertion takes based on rightmost traversal so we reverse the list 
        Collections.reverse(list);
        return list;
    }
}

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter size");
        int n = sc.nextInt();
        int arr[] = new int[n];
        for(int i=0;i<n;i++){
            arr[i] = sc.nextInt();
        }
        HashSet<Integer> res = removeDups(arr);
        System.out.println(res);
        int val = removeDuplicates(arr);
        System.out.println();
        for(int i=0; i<val;i++){
            System.out.print(arr[i]+" ");
        }
    }
    private static HashSet<Integer> removeDups(int arr[]){
        // hashset doesnot allow duplicates so by traversing the array we add each element to the set and then we return it 
	// but here tc = n and sc = n
        HashSet<Integer> list = new HashSet<>();
        int n = arr.length;
        for(int j:arr){
            list.add(j);
        }
        return list;
    }
    private static int removeDuplicates(int arr[]){
        int n = arr.length-1;
        if(n<=1){
            return n;
        }
        // here index indicates the indices of element of occurence >> 
        //[1,2,2,3,3,4,5] >> 1,2,3,4,5
        // we update values from second index because fst ele is consider as unique because it will be consider as fst occurence
        int index = 1;
        // the ele is compared with the previous elements as the elements are in sorted order index 
        // indicates the fst occurence 
        // so the elements are not deleted or removed but those are replaced until the unique ele and then those are printed upto that only 
        for(int i=1;i<n;i++){
            if(arr[i]!=arr[i-1]){
                arr[index] = arr[i];
                index++;
            }
        }
        return index;
    }
}
